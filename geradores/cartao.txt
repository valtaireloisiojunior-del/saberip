function gerarNumeroCartao() {
  let cartao = [4]; // VISA
  for (let i = 1; i < 15; i++) cartao.push(Math.floor(Math.random() * 10));

  let soma = 0;
  for (let i = 0; i < cartao.length; i++) {
    let num = cartao[cartao.length - 1 - i];
    if (i % 2 === 0) {
      num *= 2;
      if (num > 9) num -= 9;
    }
    soma += num;
  }

}
